\subsection{Examples}
\begin{namedframe}{Odd Number Squared}
	\begin{example}
		Given an odd integer $N$, prove that $N^2$ is also an odd integer.	
	\end{example}
	\pause
	If $N$ is odd, it can be expressed as $2M-1$.
	\pause
	\begin{align*}
		\uncover<+->{N^2 &= (2M-1)^2\\}
		\uncover<+->{ &= 4M^2-4M+1\\}
		\uncover<+->{ &= 2(2M^2-4M)+1}
	\end{align*}
	\uncover<+->{
	Therefore $N^2$ is also odd.}	
\end{namedframe}
