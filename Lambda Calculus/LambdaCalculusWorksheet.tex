	\section*{Definitions}
		\begin{minipage}{0.5\textwidth}
			\subsection*{Booleans}
			$\TRUE = \lambda xy.x$\\
			$\FALSE = \lambda xy.y$\\
			$\NOT = \lambda b . b (\FALSE\ \TRUE)$\\
			$\AND = (\lambda pq . p) (q\ p)$\\
			$\IFTHENELSE = (\lambda btf . b) (t\ f)$
		\end{minipage}
		\begin{minipage}{0.5\textwidth}
			\subsection*{Church Numerals}
			$0 = \lambda fx.x$\\
			$1 = \lambda fx.f(x)$\\
			$2 = \lambda fx.f(f(x))$\\
			$n = \lambda fx.f^n (x)$\\
			$\SUCC = \lambda nfx.f (nfx)$
		\end{minipage}
