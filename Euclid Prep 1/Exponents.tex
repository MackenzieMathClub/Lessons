\section{Exponents}
\subsection{Review}
\begin{namedframe}{Formulas}
	When $a, b, x, y \in \real$ and $n \in \integer \such n \neq 0$:
	\begin{columns}
		\begin{column}{0.5\textwidth}
			\begin{align*}
				a^{\frac{1}{n}} &= \sqrt[n]{a}\\
				a^0             &= 1 \text{ if } a \neq 0\\
				a^{-x}          &= \frac{1}{a^x} \text{ if } a \neq 0\\
				\frac{a^x}{b^x} &= \left( \frac{a}{b} \right)^x \text{ if } b \neq 0
			\end{align*}
		\end{column}
		\begin{column}{0.5\textwidth}
			\begin{align*}
				\frac{a^x}{a^y}       &= a^{x-y} \text{ if } a \neq 0\\
				\left( a^x \right)^ y &= a^{xy}\\
				a^x \cdot b^x         &= (ab)^x\\
				a^x a^y               &= a^{x + y}
			\end{align*}
		\end{column}
	\end{columns}
	$0^0$ is not defined.
\end{namedframe}
\subsection{Exponent problems}
\begin{namedframe}{Exponents problem 1}
	\begin{exampleblock}{Problem}
		If $m$ and $k$ are integers, find all solutions to the equation:
		\vspace{-1ex}
		\[9(7^k + k^{k+2}) = 5^{m+3} + 5^m\]
		\vspace{-4ex}
	\end{exampleblock}
	\pause
	\begin{block}{Solution}
		\vspace{-3ex}
		\begin{align*}
			\uncover<+->{9(1 + 7^2)7^k &= 5^m(5^3 + 1)\\}
			\uncover<+->{(450)7^k &= 5^m(126)\\}
			\uncover<+->{(2 \cdot 3^2 \cdot 5^2)7^k &= 5^m(2 \cdot 3^2 \cdot 7)\\}
			\uncover<+->{5^2 \cdot 7^k &= 5^m \cdot 7}
		\end{align*}
		\uncover<+->{Since the integer factorization of numbers is always unique and both $m$ and $k$ are integers:}
		\uncover<+->{$m = 2$ and $k = 1$.}
	\end{block}
\end{namedframe}
\begin{namedframe}{Exponents problem 2}
	\begin{exampleblock}{Problem}
		The graph of $y = m^x$ passed through the points $(2,5)$ and $(5,n)$.
		What is the value of $mn$?
	\end{exampleblock}
	\pause
	\begin{block}{Solution}
		We know that $m^2 = 5$ and $n = m^5$.
		The solution is trivial from here:
		\begin{align*}
			\uncover<+->{m &= \pm\sqrt{5}\\}
			\uncover<+->{n &= (\pm\sqrt{5})^5\\}
			\uncover<+->{mn &= \uncover<+->{\left( \pm\sqrt{5} \right)^6} = \uncover<+->{\left( \sqrt{5} \right)^6} = \uncover<+->{\left( \sqrt{5} \cdot \sqrt{5} \right)^3} = \uncover<+->{5^3} = \uncover<+->{125}}
		\end{align*}
	\end{block}
\end{namedframe}
